146  !> @brief Add neighbors and nbrs-of-nbrs to the model tree
147  !<
148  recursive subroutine addmodelneighbors(this, model_id, &
149  virtual_exchanges, &
150  depth, mask)
151  use virtualexchangemodule, only: get_virtual_exchange
152  class(spatialmodelconnectiontype) :: this !< this connection
153  integer(I4B) :: model_id !< the model (id) to add neighbors for
154  type(listtype) :: virtual_exchanges !< list with all virtual exchanges
155  integer(I4B), value :: depth !< the maximal number of exchanges between
156  integer(I4B), optional :: mask !< don't add this one as a neighbor
157  ! local
158  integer(I4B) :: i, n
159  class(virtualexchangetype), pointer :: v_exg
160  integer(I4B) :: neighbor_id
161  integer(I4B) :: model_mask
162  type(stlvecint) :: nbr_models
163 
164  if (.not. present(mask)) then
165  model_mask = 0
166  else
167  model_mask = mask
168  end if
169 
170  call nbr_models%init()
171 
172  ! first find all direct neighbors of the model and add them,
173  ! avoiding duplicates
174  do i = 1, virtual_exchanges%Count()
175  neighbor_id = -1
176  v_exg => get_virtual_exchange_from_list(virtual_exchanges, i)
177  if (v_exg%v_model1%id == model_id) then
178  neighbor_id = v_exg%v_model2%id
179  else if (v_exg%v_model2%id == model_id) then
180  neighbor_id = v_exg%v_model1%id
181  end if
182 
183  ! check if there is a neighbor, and it is not masked
184  ! (to prevent back-and-forth connections)
185  if (neighbor_id > 0) then
186 
187  ! check if masked
188  if (neighbor_id == model_mask) cycle
189  call nbr_models%push_back_unique(neighbor_id)
190  call this%halo_models%push_back_unique(neighbor_id)
191  call this%halo_exchanges%push_back_unique(v_exg%id)
192  end if
193 
194  end do
195 
196  depth = depth - 1
197  if (depth == 0) then
198  ! and we're done with this branch
199  call nbr_models%destroy()
200  return
201  end if
202 
203  ! now recurse on the neighbors up to the specified depth
204  do n = 1, nbr_models%size
205  call this%addModelNeighbors(nbr_models%at(n), virtual_exchanges, &
206  depth, model_id)
207  end do
208 
209  ! we're done with the tree
210  call nbr_models%destroy()
211 
212  end subroutine addmodelneighbors

304  !> @brief Mask the owner's connections
305  !!
306  !! Determine which connections are handled by the interface model
307  !! (using the connections object in its discretization) and
308  !< set their mask to zero for the owning model.
309  subroutine maskownerconnections(this)
310  use csrutilsmodule, only: getcsrindex
311  class(spatialmodelconnectiontype) :: this !< the connection
312  ! local
313  integer(I4B) :: ipos, n, m, nloc, mloc, csr_idx
314  type(connectionstype), pointer :: conn
315 
316  ! set the mask on connections that are calculated by the interface model
317  conn => this%interface_model%dis%con
318  do n = 1, conn%nodes
319  ! only for connections internal to the owning model
320  if (.not. this%ig_builder%idxToGlobal(n)%v_model == this%owner) then
321  cycle
322  end if
323  nloc = this%ig_builder%idxToGlobal(n)%index
324 
325  do ipos = conn%ia(n) + 1, conn%ia(n + 1) - 1
326  m = conn%ja(ipos)
327  if (.not. this%ig_builder%idxToGlobal(m)%v_model == this%owner) then
328  cycle
329  end if
330  mloc = this%ig_builder%idxToGlobal(m)%index
331 
332  if (conn%mask(ipos) > 0) then
333  ! calculated by interface model, set local model's mask to zero
334  csr_idx = getcsrindex(nloc, mloc, this%owner%ia, this%owner%ja)
335  if (csr_idx == -1) then
336  ! this can only happen with periodic boundary conditions,
337  ! then there is no need to set the mask
338  if (this%ig_builder%isPeriodic(nloc, mloc)) cycle
339 
340  write (*, *) 'Error: cannot find cell connection in global system'
341  call ustop()
342  end if
343 
344  if (this%owner%dis%con%mask(csr_idx) > 0) then
345  call this%owner%dis%con%set_mask(csr_idx, 0)
346  else
347  ! edge case, someone will be calculating this connection
348  ! so we ignore it here
349  write (*, *) 'Warning: overlap detected, no mask on connection ', &
350  nloc, ':', mloc, ' in model ', trim(this%owner%name), &
351  ' for Exchange ', trim(this%prim_exchange%name)
352  call conn%set_mask(ipos, 0)
353  end if
354  end if
355  end do
356  end do
357 
358  end subroutine maskownerconnections
