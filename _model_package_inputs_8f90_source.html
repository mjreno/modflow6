379  !> @brief create the package type list
380  !<
381  subroutine modelpkgs_create(this, ftypes)
382  ! -- modules
383  use sortmodule, only: qsort
384  ! -- dummy
385  class(modelpackageinputstype) :: this
386  type(characterstringtype), dimension(:), contiguous, &
387  pointer :: ftypes
388  ! -- local
389  integer(I4B), dimension(:), allocatable :: cunit_idxs, indx
390  character(len=LENPACKAGETYPE) :: ftype
391  integer(I4B) :: n, m
392  logical(LGP) :: found
393  character(len=LINELENGTH) :: errmsg
394  !
395  ! -- allocate
396  allocate (cunit_idxs(0))
397  !
398  ! -- identify input packages and check that each is supported
399  do n = 1, size(ftypes)
400  !
401  ! -- type from model name file packages block
402  ftype = ftypes(n)
403  found = .false.
404  !
405  ! -- search supported types for this filetype
406  do m = 1, this%niunit
407  if (this%cunit(m) == ftype) then
408  ! -- set found
409  found = .true.
410  !
411  ! -- add to cunit list if first instance of this type
412  if (any(cunit_idxs == m)) then
413  ! no-op
414  else
415  call expandarray(cunit_idxs)
416  cunit_idxs(size(cunit_idxs)) = m
417  end if
418  !
419  ! -- exit search
420  exit
421  end if
422  end do
423  !
424  ! -- set error if namfile pkg filetype is not supported
425  if (.not. found) then
426  write (errmsg, '(a,a,a,a,a)') 'Model package type not supported &
427  &[model=', trim(this%modelname), ', type=', &
428  trim(ftype), '].'
429  call store_error(errmsg)
430  call store_error_filename(this%modelfname)
431  end if
432  end do
433  !
434  ! -- allocate the pkglist
435  allocate (this%pkglist(size(cunit_idxs)))
436  !
437  ! -- sort cunit indexes
438  allocate (indx(size(cunit_idxs)))
439  call qsort(indx, cunit_idxs)
440  !
441  ! -- create sorted LoadablePackageType object list
442  do n = 1, size(cunit_idxs)
443  call this%pkglist(n)%create(this%modelname, this%cunit(cunit_idxs(n)))
444  end do
445  !
446  ! -- cleanup
447  deallocate (cunit_idxs)
448  deallocate (indx)
449  !
450  ! -- return
451  return
452  end subroutine modelpkgs_create
