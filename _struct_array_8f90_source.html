487  !> @brief reallocate local memory for deferred vectors if necessary
488  !<
489  subroutine check_reallocate(this)
490  class(structarraytype) :: this !< StructArrayType
491  integer(I4B) :: i, j, newsize
492  integer(I4B), dimension(:), pointer, contiguous :: p_int1d
493  real(DP), dimension(:), pointer, contiguous :: p_dbl1d
494  type(characterstringtype), dimension(:), pointer, contiguous :: p_charstr1d
495  integer(I4B) :: reallocate_mult
496  !
497  ! -- set growth rate
498  reallocate_mult = 2
499  !
500  do j = 1, this%ncol
501  !
502  ! -- reallocate based on memtype
503  select case (this%struct_vector_1d(j)%memtype)
504  !
505  case (1) ! -- memtype integer
506  !
507  ! -- check if more space needed
508  if (this%nrow > this%struct_vector_1d(j)%size) then
509  !
510  ! -- calculate new size
511  newsize = this%struct_vector_1d(j)%size * reallocate_mult
512  !
513  ! -- allocate new vector
514  allocate (p_int1d(newsize))
515  !
516  ! -- copy from old to new
517  do i = 1, this%struct_vector_1d(j)%size
518  p_int1d(i) = this%struct_vector_1d(j)%int1d(i)
519  end do
520  !
521  ! -- deallocate old vector
522  deallocate (this%struct_vector_1d(j)%int1d)
523  !
524  ! -- update struct array object
525  this%struct_vector_1d(j)%int1d => p_int1d
526  this%struct_vector_1d(j)%size = newsize
527  end if
528  !
529  case (2) ! -- memtype real
530  if (this%nrow > this%struct_vector_1d(j)%size) then
531  !
532  newsize = this%struct_vector_1d(j)%size * reallocate_mult
533  !
534  allocate (p_dbl1d(newsize))
535  !
536  do i = 1, this%struct_vector_1d(j)%size
537  p_dbl1d(i) = this%struct_vector_1d(j)%dbl1d(i)
538  end do
539  !
540  deallocate (this%struct_vector_1d(j)%dbl1d)
541  !
542  this%struct_vector_1d(j)%dbl1d => p_dbl1d
543  this%struct_vector_1d(j)%size = newsize
544  end if
545  !
546  case (3) ! -- memtype charstring
547  if (this%nrow > this%struct_vector_1d(j)%size) then
548  !
549  newsize = this%struct_vector_1d(j)%size * reallocate_mult
550  !

570  !> @brief read from the block parser to fill the StructArrayType
571  !<
572  subroutine read_from_parser(this, parser, iout)
573  class(structarraytype) :: this !< StructArrayType
574  type(blockparsertype) :: parser !< block parser to read from
575  integer(I4B), intent(in) :: iout !< unit number for output
576  logical(LGP) :: endOfBlock
577  integer(I4B) :: irow, j, k
578  integer(I4B) :: intval, numval
579  character(len=LINELENGTH) :: str
580  character(len=:), allocatable :: line
581  !
582  ! -- initialize index irow
583  irow = 0
584  !
585  ! -- read entire block
586  do
587  !
588  ! -- read next line
589  call parser%GetNextLine(endofblock)
590  !
591  if (endofblock) then
592  ! -- no more lines
593  exit
594  !
595  else if (this%deferred_shape) then
596  !
597  ! -- shape unknown, track lines read
598  this%nrow = this%nrow + 1
599  !
600  ! -- check and update memory allocation
601  call this%check_reallocate()
602  end if
603  !
604  ! -- update irow index
605  irow = irow + 1
606  !
607  ! -- handle line reads by column memtype
608  do j = 1, this%ncol
609  !
610  select case (this%struct_vector_1d(j)%memtype)
611  !
612  case (1) ! -- memtype integer
613  !
614  ! -- if reloadable block and first col, store blocknum
615  if (j == 1 .and. this%blocknum > 0) then
616  ! -- store blocknum
617  this%struct_vector_1d(j)%int1d(irow) = this%blocknum
618  else
619  ! -- read and store int
620  this%struct_vector_1d(j)%int1d(irow) = parser%GetInteger()
621  end if
622  !
623  case (2) ! -- memtype real
624  !
625  this%struct_vector_1d(j)%dbl1d(irow) = parser%GetDouble()
626  !
627  case (3) ! -- memtype charstring
